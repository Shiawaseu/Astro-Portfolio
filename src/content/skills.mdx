import { Badge } from "../components/ui/badge.tsx"; 

export const skills = [
  "C#",
  "TypeScript",
  "JavaScript",
  "Node.js",
  "React",
  "Luau",
  "Python",
  "HTML5",
  "NPM",
  "PNPM",
  "Astro",
  "Next.js",
  "Tailwind CSS",
  "Supabase",
  "Prisma",
  "Redis",
  "VS Code",
  "Visual Studio",
];

export default function Skills({ names }) {
  if (!names) return;
  const filteredSkills = names === "all"
    ? skills
    : skills.filter(skill => names.split(',').map(name => name.trim()).includes(skill.toLowerCase()));

  return (
    <div className="space-x-1">
      {filteredSkills.map((skill, index) => (
        <Badge key={index} variant="default">
          <p className="font-extrabold">{skill}</p>
        </Badge>
      ))}
    </div>
  );
}
