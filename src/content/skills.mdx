export default function Skills({ names, showIcons = false }) { // Disabled for now
  if (!names) return;
  const filteredSkills = names === "all"
    ? skills
    : skills.filter(skill => names.split(',').map(name => name.trim()).includes(skill.name.toLowerCase()));

  return (
    <div className="space-x-1">
      {filteredSkills.map((skill, index) => (
        <Badge key={index} variant="default">
          {showIcons && skill.icon}
          <p className={`font-bold ${showIcons ? 'ml-1' : ''}`}>{skill.name}</p>
        </Badge>
      ))}
    </div>
  );
}
